              function countBs(bString) {

              var count = 0;

              for (var i = 0; i < bString.length; i++) {

              if (bString.charAt(i) == 'B') {

              count = (count + 1);

              }

              function sum(array) {
              var sm = 0;
              for (var i = 0; i < array.length; i++) {
              sm += parseInt(array[i]);
              }
              return sm;
              }
